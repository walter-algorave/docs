`make_snapshot()` vive in `helpers.py` ed è importata direttamente nei test (`from helpers import make_snapshot`). `conftest.py` non è importabile come modulo Python normale — contiene solo le fixture pytest.

```python
# helpers.py — helper condiviso tra tutti i file di test
def make_snapshot(**kwargs) -> dict:
    base = {
        "game_state":              "Preflop",
        "player_cards":            [],
        "table_cards":             [],
        "relative_position":       kwargs.get("position", "CO"),
        "active_players":          kwargs.get("active", [0, 1, 2, 3, 4, 5]),
        "initial_active_players":  kwargs.get("active", [0, 1, 2, 3, 4, 5]),
        "hand_id":                 kwargs.get("hand_id", 1),
        "player_bets":             kwargs.get("bets", {}),
        "pot_amount":              kwargs.get("pot", 0.0),
        "hero_folded":             kwargs.get("hero_folded", False),
        "hero_had_cards":          kwargs.get("hero_had_cards", True),
        "dealer_position":         kwargs.get("dealer", 0),
        "table_name":              kwargs.get("table_name", "T1"),
    }
    base.update({k: v for k, v in kwargs.items() if k in base or k == "cards"})
    if "cards" in kwargs:
        base["player_cards"] = kwargs["cards"]
    return base
```
